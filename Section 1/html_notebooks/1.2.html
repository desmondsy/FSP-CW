snp_close_arma.columns = ['True']

snp_close_arma['Res'] = arma_model_all.resid
snp_close_arma['Prediction'] = snp_close_arma['True'] - snp_close_arma['Res']

arma_model = ARIMA(snp_close_train,order=(1,0,0)).fit()

snp_close_train['Prediction'] = snp_close_train['True'] - snp_close_train['Res']
#forecast
fc,_,_ = arma_model.forecast(306, alpha=0.05)
fc_series = pd.Series(fc)
