tickers = ['CAG','MAR','LIN','HCP','MAT']
stocks = df[tickers].dropna()
stocks_ma = stocks.rolling(window=66,min_periods=66).mean()
stocks_detrended = stocks.sub(stocks_ma).dropna()
log_returns = stocks.apply(np.log).diff().dropna()

#correlations
price_corr = stocks.corr().sum().sort_values(ascending=True).index.values
price_corr = stocks[price_corr].corr()
log_ret_corr= log_returns.corr().sum().sort_values(ascending=True).index.values
log_ret_corr = log_returns[log_ret_corr].corr()

plt.figure(figsize=(17,4))
plt.subplot(1,2,1)
plt.plot(stocks)
plt.xlabel('Date')
plt.ylabel('Price')
plt.title('Time series of 5 Stocks')
plt.grid('major',axis='both')
plt.legend(tickers)
