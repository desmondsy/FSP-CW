def adaptiveMarkowitz(df,window):
    M = df.shape[1] # number of stocks
    e = np.ones((M,1))
    adaptive_w = np.zeros((df.shape[0]-window,M))
    rets = pd.DataFrame(index=df.iloc[window:,:].index)
    w_equal = np.dot(e,1/M)
    var = []
    
    # adaptive weights
    for i in range(df.shape[0]-window):
        r = df.iloc[i:window+i,:]
        mu = r.mean()
        C = np.dot((r-mu).T,r-mu)/window
        C_inv = np.linalg.inv(C)
        adaptive_w[[i],:] = (np.dot(C_inv,e) / np.dot(np.dot(e.T,C_inv),e)).T
        var.append(1/C_inv.sum().sum())
    rets['adaptive w ret'] = np.sum(np.multiply(df.iloc[window:,:],adaptive_w),axis=1)
    
    # portfolio returns with equal weights
    rets['equal w ret'] = np.dot(df.iloc[window:,:],w_equal)
    
    # portfolio returns with non-adaptive optimal weights
    cov = df.iloc[window:,:].corr()
    non_adaptive_w = np.dot(np.linalg.inv(cov),e) / np.dot(np.dot(e.T,np.linalg.inv(cov)),e)
    rets['optimal w ret'] = np.dot(df.iloc[window:,:],non_adaptive_w)
    
    var_df = np.mean(var), 1/np.linalg.inv(cov).sum().sum(), np.dot(np.dot(w_equal.T,cov),w_equal)
    
    return adaptive_w, rets, var, var_df
